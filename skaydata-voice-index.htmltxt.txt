  <script>
    // ⚠️【超重要】お前のMakeのWebhook URLをここに貼り付けろ！！
    const MAKE_WEBHOOK_URL = "ここにMakeのWebhookのURLを貼り付ける";

    const micButton = document.getElementById('micButton');
    const statusText = document.getElementById('status');

    // 音声認識の設定 (iPhoneのSafariにも対応させるため webkit を使用)
    const SpeechRecognition = window.SpeechRecognition || window.webkitSpeechRecognition;
    
    if (!SpeechRecognition) {
      statusText.innerHTML = "※このブラウザは音声認識に未対応です。<br>SafariかChromeを使用してください。";
      micButton.disabled = true;
    } else {
      const recognition = new SpeechRecognition();
      recognition.lang = 'ja-JP';
      recognition.interimResults = false;
      recognition.maxAlternatives = 1;

      micButton.addEventListener('click', () => {
        recognition.start();
      });

      recognition.onstart = () => {
        micButton.classList.add('recording');
        statusText.textContent = "音声を聞き取っています...";
      };

      recognition.onspeechend = () => {
        recognition.stop();
        micButton.classList.remove('recording');
        statusText.textContent = "データ送信中...";
      };

      recognition.onresult = (event) => {
        const speechResult = event.results[0][0].transcript;
        statusText.textContent = `「${speechResult}」を送信中...`;
        
        // MakeのWebhookへデータを送信 (POST)
        sendToMake(speechResult);
      };

      recognition.onerror = (event) => {
        micButton.classList.remove('recording');
        statusText.textContent = `エラーが発生しました: ${event.error}`;
      };
    }

    // Makeにデータを送る関数
    function sendToMake(textData) {
      fetch(MAKE_WEBHOOK_URL, {
        method: "POST",
        headers: {
          "Content-Type": "application/json"
        },
        body: JSON.stringify({ text: textData }) // Make側は「text」という名前で受け取る
      })
      .then(response => {
        if(response.ok) {
          statusText.textContent = "✅ 送信完了！";
          setTimeout(() => { statusText.textContent = "ボタンを押して話し始めてください"; }, 3000);
        } else {
          statusText.textContent = "❌ 送信に失敗しました";
        }
      })
      .catch(error => {
        statusText.textContent = "❌ ネットワークエラーです";
      });
    }
  </script>
